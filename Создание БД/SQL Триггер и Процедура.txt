CREATE OR REPLACE FUNCTION fn_check_stock_operation()
RETURNS TRIGGER AS $$
DECLARE
    v_expiration_date DATE;
    v_purchase_price DECIMAL(10, 2);
    v_current_quantity INT;
    v_op_date TIMESTAMP;
BEGIN
    v_op_date := COALESCE(NEW.operation_date, CURRENT_TIMESTAMP);

    SELECT expiration_date, purchase_price, current_quantity 
    INTO v_expiration_date, v_purchase_price, v_current_quantity
    FROM batches 
    WHERE id = NEW.batch_id;

    IF NOT FOUND THEN
        RAISE EXCEPTION 'Партия с ID % не найдена', NEW.batch_id;
    END IF;

    IF NEW.operation_type = 'выдача' AND v_op_date::DATE > v_expiration_date THEN
        RAISE EXCEPTION 'ВНИМАНИЕ: Выдача заблокирована! Срок годности препарата ИСТЕК!';
    END IF;

    IF NEW.operation_type IN ('выдача', 'утилизация') THEN
        IF NEW.quantity > v_current_quantity THEN
            RAISE EXCEPTION 'Недостаточное количество препарата в партии. Доступно: %, Запрошено: %', v_current_quantity, NEW.quantity;
        END IF;
    END IF;

    NEW.total_cost := NEW.quantity * v_purchase_price;
    NEW.operation_date := v_op_date;

    IF NEW.operation_type = 'приход' THEN
        UPDATE batches 
        SET current_quantity = current_quantity + NEW.quantity 
        WHERE id = NEW.batch_id;
    ELSE
        UPDATE batches 
        SET current_quantity = current_quantity - NEW.quantity 
        WHERE id = NEW.batch_id;
    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE OR REPLACE TRIGGER trg_before_stock_operation
BEFORE INSERT ON stock_operations
FOR EACH ROW
EXECUTE FUNCTION fn_check_stock_operation();


CREATE OR REPLACE PROCEDURE write_off_expired_batch(
    p_batch_id INT, 
    p_employee_tab_number VARCHAR
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_current_quantity INT;
    v_department_id INT;
BEGIN
    SELECT current_quantity 
    INTO v_current_quantity
    FROM batches 
    WHERE id = p_batch_id;

    IF NOT FOUND THEN
        RAISE EXCEPTION 'Партия с ID % не найдена', p_batch_id;
    END IF;

    IF v_current_quantity > 0 THEN
        SELECT department_id INTO v_department_id 
        FROM system_users 
        WHERE tab_number = p_employee_tab_number;
        
        IF NOT FOUND THEN
            RAISE EXCEPTION 'Сотрудник с табельным номером % не найден', p_employee_tab_number;
        END IF;

        INSERT INTO stock_operations (batch_id, employee_id, department_id, operation_type, quantity)
        VALUES (p_batch_id, p_employee_tab_number, v_department_id, 'утилизация', v_current_quantity);
    END IF;
END;
$$;
